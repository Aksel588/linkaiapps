Usage Example 
from aion.text import count_words, count_lines, summarize_text, extract_emails, extract_urls, highlight_keywords

sample_text = """
Hello, this is LinkAI-Aion!
Contact: aksel@example.com
Website: https://linkaiapps.com
This is a multi-line sample for testing.
"""

# Count words in the sample
print("Word count:", count_words(sample_text))

# Count lines in the sample
print("Line count:", count_lines(sample_text))

# Summarize the sample (first 2 lines)
print("Summary:", summarize_text(sample_text, max_lines=2))

# Extract emails
print("Emails:", extract_emails(sample_text))

# Extract URLs
print("URLs:", extract_urls(sample_text))

# Highlight keywords
print("Highlighted:", highlight_keywords(sample_text, ["LinkAI-Aion", "test"]))
